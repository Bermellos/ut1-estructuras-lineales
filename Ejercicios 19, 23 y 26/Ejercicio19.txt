    METODO controlCorchetes(listaDeEntrada: Lista de Caracteres) : Booleano

        DEFINIR pila COMO Pila de Caracteres
        pila <- nueva Pila

        PARA CADA caracter EN listaDeEntrada HACER      --> complejidad O(n) ya que es un for que recorre toda la lista

            SI caracter = '{' ENTONCES
                APILAR pila, caracter
            SINO
                SI caracter = '}' ENTONCES

                    SI pila está vacía ENTONCES
                        RETORNAR FALSO
                    SINO
                        DESAPILAR pila
                    FIN SI

                FIN SI
            FIN SI

        FIN PARA

        SI pila está vacía ENTONCES
            RETORNAR VERDADERO
        SINO
            RETORNAR FALSO
        FIN SI

    FIN METODO
