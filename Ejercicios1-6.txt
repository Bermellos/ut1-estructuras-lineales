1)a) El tiempo de ejecución de este algoritmo es O(n)

public static int enRango (int[] a, int bajo, int alto) 
{ 
   int contador = 0; 
   for (int i = 0; i < a.length; i++) ---> O(n)
   { 
     if (a[i] >= bajo && a[i] < alto) 
       contador++; 
    } 
    return contador; 
}
